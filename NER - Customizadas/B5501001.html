 *

JDEBFRTN (ID) JDEBFWINAPI ValidacaoCtaBanrisul (LPBHVRCOM lpBhvrCom, LPVOID lpVoid, LPDSD5501001 lpDS)  
{
   /************************************************************************
    *  Variable declarations
    ************************************************************************/

   /************************************************************************
    * Declare structures
    ************************************************************************/
	
	int iConta[NUM_DIGITOS];
	int	iCont=0;					/* Contador */
	/*int iDigito=0;*/				/* Digito Verificador */
	int iTemp=0;					/* Variavel Temporaria */
	int iSum=0;						/* Acumulador */
	int iResto=0;					/* Resto da divisao por 11 */

   /************************************************************************
    * Declare pointers
    ************************************************************************/

   /************************************************************************
    * Check for NULL pointers
    ************************************************************************/
   if ((lpBhvrCom == (LPBHVRCOM) NULL) ||
       (lpVoid    == (LPVOID)    NULL) ||
       (lpDS      == (LPDSD5501001)	NULL))
   {
     jdeErrorSet (lpBhvrCom, lpVoid, (ID) 0, _J("4363"), (LPVOID) NULL);
     return ER_ERROR;
   }

   /************************************************************************
    * Set pointers
    ************************************************************************/

   /************************************************************************
    * Main Processing
    ************************************************************************/


   /* Inicializa estrutura de Digito de Controle */
   memset ((void *)(&lpDS->szControlDigit), (int)(_J('\0')), sizeof(2));

	/* Converter caracteres para digitos */
   for ( iCont = 0; iCont <= NUM_DIGITOS; iCont++)
	{
	   iConta[iCont] = (int) (lpDS->szCustBankAcctNumber[iCont])-(int)_J('0');
	
    if ((iConta[iCont] > 9) || (iConta[iCont] < 0))
		{
		jdeErrorSet (lpBhvrCom, lpVoid, (ID) 0, _J("0002"), (LPVOID) NULL);
		return ER_ERROR;
		}  /* if iCont > 9 ... */
	}  /* for iCont */

	/* Trocar posicoes dos digitos */
   iTemp = iConta[2];
   iConta[2] = iConta[1];
   iConta[1] = iConta[0];
   iConta[0] = iTemp;

	/* Calcular o digito */
   /* iSum = 0;*/
   iSum = iConta[0]*7 + iConta[1]*8 + iConta[2]*9 + iConta[3]*4 + iConta[4]*5 + iConta[5]*6 + iConta[6]*7 + iConta[7]*8 + iConta[8]*9;
   iResto = (iSum% 11);

   /* Converter digito verificador */
   if (iResto == 10)
   {
	lpDS->szControlDigit[0] = _J('6');
   }
   else
   {
	lpDS->szControlDigit[0] = (JCHAR)(iResto + 48);
   }

   /************************************************************************
    * Function Clean Up
    ************************************************************************/

   return (ER_SUCCESS);
}
