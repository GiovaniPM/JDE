/**************************************************************************
 *  Business Function:  WriteDataOutputFile5000
 *
 *        Description:  Write Data Output File
 *
 *         Parameters:
 *           LPBHVRCOM           lpBhvrCom    Business Function Communications
 *           LPVOID              lpVoid       Void Parameter - DO NOT USE!
 *           LPDSD5576B03        lpDS         Parameter Data Structure Pointer  
 *
 *************************************************************************/

JDEBFRTN (ID) JDEBFWINAPI WriteDataOutputFile1500 (LPBHVRCOM lpBhvrCom, LPVOID lpVoid, LPDSD5576B03 lpDS)  
{
   /************************************************************************
    *  Variable declarations
    ************************************************************************/
   ID  idReturn = ER_SUCCESS; /* Status Code. */
   int iLastPosition; /* Last position of the data which will be saved into the magnetic file. */
   
   /************************************************************************
    * Declare structures
    ************************************************************************/

   /************************************************************************
    * Declare pointers
    ************************************************************************/
   FILE *pOutputMagneticFile = (FILE *) NULL; /* Pointer to the Magnetic file. */

   /************************************************************************
    * Check for NULL pointers
    ************************************************************************/
   if ((lpBhvrCom == (LPBHVRCOM) NULL) ||
       (lpVoid    == (LPVOID)    NULL) ||
       (lpDS      == (LPDSD5576B03)	NULL))
   {
     jdeErrorSet (lpBhvrCom, lpVoid, (ID) 0, _J("4363"), (LPVOID) NULL);
     return ER_ERROR;
   }

   /************************************************************************
    * Set pointers
    ************************************************************************/

   /************************************************************************
    * Main Processing
    ************************************************************************/
/*    Check for original flat file name */

   if ( IsStringBlank(lpDS->szNameFlatFile) )
   {

/*    No name was sent to this function. */

      jdeErrorSet (lpBhvrCom, lpVoid, IDERRszNameFlatFile_1, _J("0003"),
                   (LPVOID) NULL);
      idReturn = ER_ERROR;
   }
   else
   {
      if ( ( pOutputMagneticFile = jdeFopen(lpDS->szNameFlatFile, _J("a+")) ) ==
           (FILE *) NULL )
      {

/*    The output file could not be created. An error message is informed */

         idReturn = ER_ERROR;
      }
      else
      {

/*    The output file was opened. */
/*    Convert to integer the last position in the data. */
//****** Juliano Rach - 31/05/2011 - Modificado trecho do codigo por causa do Warning de incompatibilidade. ******
//warning C4133: 'function' : incompatible types - from 'ZCHAR [33]' to 'const JCHAR *'
//       iLastPosition = jdeAtoi ( lpDS->mnRecordSize.String );
         iLastPosition = atoi ( lpDS->mnRecordSize.String );

/*    Set the endo of the data. The end of the data match with the size parameter. */

         lpDS->szText[iLastPosition] = _J('\0');

/*    Write the data into the Magnetic File. */
/*		 if (jdeStrstr(lpDS->szText,_J("\r")) == NULL)
		 {
			jdeFprintf ( pOutputMagneticFile, _J("%ls\n"), lpDS->szText );
		 }
		 else
		 {
			jdeFprintf ( pOutputMagneticFile, _J("%ls\r\n"), lpDS->szText );
		 }
*/
		#if defined (JDENV_PC)
		    jdeFprintf ( pOutputMagneticFile, _J("%ls\n"), lpDS->szText );			
		#else
		    jdeFprintf ( pOutputMagneticFile, _J("%ls\r\n"), lpDS->szText );			
		#endif

/*       fwrite ( lpDS->szText, sizeof ( char ), uNumCharacters, pOutputMagneticFile ); */

/*    Close the Magnetic File. */

         jdeFclose ( pOutputMagneticFile );
      }
   }

   /************************************************************************
    * Function Clean Up
    ************************************************************************/

   return (ER_SUCCESS);
}


/**************************************************************************
 *  Business Function:  FetchNextLineFromFlatFile1500
 *
 *        Description:  Fetch Next Line from Flat File 5000 bytes
 *
 *         Parameters:
 *           LPBHVRCOM           lpBhvrCom    Business Function Communications
 *           LPVOID              lpVoid       Void Parameter - DO NOT USE!
 *           LPDSD5576B03B       lpDS         Parameter Data Structure Pointer  
 *
 *************************************************************************/

JDEBFRTN (ID) JDEBFWINAPI FetchNextLineFromFlatFile1500 (LPBHVRCOM lpBhvrCom, LPVOID lpVoid, LPDSD5576B03B lpDS)  
{
   /************************************************************************
    *  Variable declarations
    ************************************************************************/
	int		nRecordLength=0;
	JCHAR		szRecord[5001];

   /************************************************************************
    * Declare structures
    ************************************************************************/

   /************************************************************************
    * Declare pointers
    ************************************************************************/
	ID       idJDEDBReturn		= (ID) 0;
	FILE     *lpdsFlatFileName	=  (FILE *) NULL;
	HUSER    hUser = (HUSER) NULL;

   /************************************************************************
    * Check for NULL pointers
    ************************************************************************/
   if ((lpBhvrCom == (LPBHVRCOM) NULL) ||
       (lpVoid    == (LPVOID)    NULL) ||
       (lpDS      == (LPDSD5576B03B)	NULL))
   {
     jdeErrorSet (lpBhvrCom, lpVoid, (ID) 0, _J("4363"), (LPVOID) NULL);
     return ER_ERROR;
   }

   /************************************************************************
    * Set pointers
    ************************************************************************/

   /************************************************************************
    * Main Processing
    ************************************************************************/
	lpDS->szFlatFileRecord[0] = _J('\0');
	jdeStrcpy (lpDS->szErrorMessageID, _J("    "));
    lpDS->cErrorCode = _J('0');


	/* Initialize Behavior Routine */
	idJDEDBReturn = JDB_InitBhvr(lpBhvrCom, &hUser,(JCHAR *) NULL, JDEDB_COMMIT_AUTO);
	if (idJDEDBReturn == (ID) JDEDB_FAILED)
	{
	  jdeSetGBRError (lpBhvrCom,lpVoid,(ID) 0,_J("3143"));
	  return (ER_ERROR);
	}


	/* Verify for FlatFile Name */
   if (IsStringBlank(lpDS->szFlatFileName))
   {
      /*  Invalid info in input parameters */
      jdeStrcpy (lpDS->szErrorMessageID, _J("4377"));
      if (lpDS->cSuppressErrorMessage != _J('1'))
      {
         jdeErrorSet (lpBhvrCom, lpVoid, (ID) 0, _J("4377"), (LPVOID) NULL);
      }
      lpDS->cErrorCode = _J('2');
      return (ER_ERROR);
   }

	#if defined (JDENV_PC)
	/* Open Flat File or Get File Pointer */
	/* ================================== */
	if ((lpDS->idFilePtr == (ID) 0))  /* Pointer to Flat File not exists */
	{
		/* Open Flat File */
		if( (lpdsFlatFileName = jdeFopen( lpDS->szFlatFileName, _J("rt") )) == NULL )
		{
			/*  Cannot open Flat file */
			jdeStrcpy (lpDS->szErrorMessageID, _J("3003"));
			if (lpDS->cSuppressErrorMessage != _J('1'))
			{
				jdeErrorSet (lpBhvrCom, lpVoid, (ID) 0, _J("3003"), (LPVOID) NULL);
			}
			lpDS->cErrorCode = _J('2');
			return (ER_ERROR);
		}
		else
		{
			/* Save Flat File pointer */
			lpDS->idFilePtr = (ID) jdeStoreDataPtr (hUser, (void *) lpdsFlatFileName);
		}
	}
	else
	{
		/* Get Flat File Pointer */
		lpdsFlatFileName = jdeRetrieveDataPtr (hUser, lpDS->idFilePtr);
	}

	/* Get Next Record from Flat File */
	MathNumericToInt(&lpDS->mnRecordLength, &nRecordLength);
	if (jdeFgets (szRecord, nRecordLength+2, lpdsFlatFileName) != NULL)
	{
		/* Record Exists, remove last char End of Line */
		memcpy(lpDS->szFlatFileRecord, szRecord, nRecordLength);
		lpDS->cErrorCode = _J('0');
	}
	else
	{
		/* End of file */
		/* ----------- */

		/* Clear pointer */
		jdeRemoveDataPtr(hUser, lpDS->idFilePtr);
		lpDS->idFilePtr = (ID) 0;
		
		/* Close File */
		jdeFclose (lpdsFlatFileName);
		lpDS->cErrorCode = _J('1');
	}
	#else
   /* Open Flat File or Get File Pointer */
	/* ================================== */
	if ((lpDS->idFilePtr == (ID) 0))  /* Pointer to Flat File not exists */
	{
		/* Open Flat File */
		if( (lpdsFlatFileName = jdeFopen( lpDS->szFlatFileName, _J("r") )) == NULL )
		{
			/*  Cannot open Flat file */
			jdeStrcpy (lpDS->szErrorMessageID, _J("3003"));
			if (lpDS->cSuppressErrorMessage != _J('1'))
			{
				jdeErrorSet (lpBhvrCom, lpVoid, (ID) 0, _J("3003"), (LPVOID) NULL);
			}
			lpDS->cErrorCode = _J('2');
			return (ER_ERROR);
		}
		else
		{
			/* Save Flat File pointer */
			lpDS->idFilePtr = (ID) jdeStoreDataPtr (hUser, (void *) lpdsFlatFileName);
		}
	}
	else
	{
		/* Get Flat File Pointer */
		lpdsFlatFileName = jdeRetrieveDataPtr (hUser, lpDS->idFilePtr);
	}

	/* Get Next Record from Flat File */
	MathNumericToInt(&lpDS->mnRecordLength, &nRecordLength);
	if (jdeFgetsConvert(lpBhvrCom, szRecord, nRecordLength+2, lpdsFlatFileName) != NULL)
	{
		/* Record Exists, remove last char End of Line */
		memcpy(lpDS->szFlatFileRecord, szRecord, nRecordLength * sizeof(JCHAR));
		lpDS->szFlatFileRecord[nRecordLength] = _J('\0');
		lpDS->cErrorCode = _J('0');
	}
	else
	{
		/* End of file */
		/* ----------- */

		/* Clear pointer */
		jdeRemoveDataPtr(hUser, lpDS->idFilePtr);
		lpDS->idFilePtr = (ID) 0;
		
		/* Close File */
		jdeFclose (lpdsFlatFileName);
		lpDS->cErrorCode = _J('1');
	}
	#endif
   /************************************************************************
    * Function Clean Up
    ************************************************************************/
	JDB_FreeBhvr (hUser);


   return (ER_SUCCESS);
}
