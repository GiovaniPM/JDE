/**************************************************************************
 *  Business Function:  CompleteWFMessageActivity
 *
 *        Description:  Complete Workflow Message Activity
 *
 *         Parameters:
 *           LPBHVRCOM           lpBhvrCom    Business Function Communications
 *           LPVOID              lpVoid       Void Parameter - DO NOT USE!
 *           LPDSD5598002        lpDS         Parameter Data Structure Pointer  
 *
 *************************************************************************/

JDEBFRTN (ID) JDEBFWINAPI CompleteWFMessageActivity (LPBHVRCOM lpBhvrCom, 
                                                     LPVOID lpVoid, 
                                                     LPDSD5598002 lpDS)  
{
   /************************************************************************
    *  Variable declarations
    ************************************************************************/
   ID             idReturnCode   = ER_SUCCESS;

   JCHAR           szApproved[]   = _J("03");
   JCHAR           szRejected[]   = _J("05");

   /************************************************************************
    * Declare structures
    ************************************************************************/
   JDEDB_RESULT   rc             = JDEDB_PASSED;

   LPJDEAPP       lpObj          = (LPJDEAPP)NULL;
   LPAPPVARIABLES pAppVars       = (LPAPPVARIABLES)NULL;
   LPGLOBALAPP    pGlblApp       = (LPGLOBALAPP)NULL;
   HUSER          hUser          = (HUSER)NULL;
   KERNEL_TYPE    eKernelType    = KERNEL_QUEUE;
   WFOBJ          wfObj          = { 0 };

   /************************************************************************
    * Declare pointers
    ************************************************************************/

   /************************************************************************
    * Check for NULL pointers
    ************************************************************************/
   if ((lpBhvrCom == (LPBHVRCOM) NULL) ||
       (lpVoid    == (LPVOID)    NULL) ||
       (lpDS      == (LPDSD5598002)   NULL))
   {
      jdeErrorSet (lpBhvrCom, lpVoid, (ID) 0, _J("4363"), (LPVOID) NULL);
      return ER_ERROR;
   }

   /************************************************************************
    * Set pointers
    ************************************************************************/
   lpObj = lpBhvrCom->lpObj;
   pAppVars = lpObj->lpzAppVars;
   pGlblApp = lpObj->lpGlobalApp;
   hUser = pGlblApp->lpHuser;
   eKernelType = JDB_GetKernelType(hUser);

   wfObj.hUser = hUser;
   wfObj.lpzAppVars = pAppVars;
   wfObj.lpBhvrCom = lpBhvrCom;
   wfObj.lpLocalSpace = (LPVOID) lpObj;

   /* The prototype for K2ExecuteWFERLineFromBSFN is located in
    * jdekprv instead of jdekprto so it is not visible here.
	* But, ResumeActiveWFProcess, will default the function pointer
	* to K2ExecuteWFERLineFromBSFN if left NULL so we won't initialize
	* this data structure member here.
	*/
   /*wfObj.pExecuteWFERLine = K2ExecuteWFERLineFromBSFN;*/
   wfObj.lpLocalSpace = lpBhvrCom;
   
   /************************************************************************
    * Main Processing
    ************************************************************************/

   if (lpDS->cApprovalActionCode == _J('A'))
   {
      rc = CompleteWFActivity(&wfObj, 
		                      lpDS->szProcessName,
                              &lpDS->mnProcessVersion,
							  &lpDS->mnProcessInstance,
                              lpDS->szActivityName,
							  &lpDS->mnActivityInstance,
                              &lpDS->mnOrganizationalGroupNumber,
                              &lpDS->mnSequenceNumber,
							  szApproved); 

	     }
   else
   {
      rc = CompleteWFActivity(&wfObj, lpDS->szProcessName,
                              &lpDS->mnProcessVersion, &lpDS->mnProcessInstance,
                              lpDS->szActivityName, &lpDS->mnActivityInstance,
                              &lpDS->mnOrganizationalGroupNumber,
                              &lpDS->mnSequenceNumber, szRejected);
   }

   ResumeActiveWFProcess(&wfObj, lpDS->szProcessName,
                         &lpDS->mnProcessVersion, &lpDS->mnProcessInstance,
                         lpDS->szActivityName);

   /************************************************************************
    * Function Clean Up
    ************************************************************************/

   return (idReturnCode);
}
