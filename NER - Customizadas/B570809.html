int I570809_CarregaArquivoMemoria(const JCHAR *NomeArquivo, JCHAR **Retorno) 
{ 
	int Tamanho = 0;
	FILE *Arquivo = jdeFopen(NomeArquivo, _J("rb"));
	if (Arquivo == NULL) 
	{ 
		*Retorno = NULL;
		return -1; 
	} 
	fseek(Arquivo, 0, SEEK_END);
	Tamanho = ftell(Arquivo);
	fseek(Arquivo, 0, SEEK_SET);
	*Retorno = (JCHAR *)malloc(Tamanho+1);
	if (Tamanho != (int)fread(*Retorno, sizeof(JCHAR), Tamanho, Arquivo)) 
	{ 
		free(*Retorno);
		return -2; 
	} 
	jdeFclose(Arquivo);
	(*Retorno)[Tamanho] = 0;
	return Tamanho;
}

		//
		jdeFwrite (ArquivoMemoria , 1 , Tamanho , Arquivo );

	}
