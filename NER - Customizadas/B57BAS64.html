void I57BAS64_CodificaBloco( BYTE Entrada[4], BYTE Saida[5], int Tamanho )
{
    Saida[0] = cb64[ Entrada[0] >> 2 ];
    Saida[1] = cb64[ ((Entrada[0] & 0x03) << 4) | ((Entrada[1] & 0xf0) >> 4) ];
    Saida[2] = (BYTE) (Tamanho > 1 ? cb64[ ((Entrada[1] & 0x0f) << 2) | ((Entrada[2] & 0xc0) >> 6) ] : ' ');
    Saida[3] = (BYTE) (Tamanho > 2 ? cb64[ Entrada[2] & 0x3f ] : ' ');
      Saida[5] = '\0';
}
