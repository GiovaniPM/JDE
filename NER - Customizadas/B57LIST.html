    ************************************************************************/
	//estrutura de lista do jde
    LINKLISTITEM                 dsItemLista;

   /************************************************************************
    * Declare pointers
    ************************************************************************/
	//nossa lista
	Lista   lpDadosLista = (void *)NULL;

   /************************************************************************
    * Check for NULL pointers
    ************************************************************************/
    if ((lpBhvrCom == (LPBHVRCOM) NULL) ||
       (lpVoid    == (LPVOID)    NULL) ||
       (lpDS      == (LPDSD57LISTC)	NULL))
    {
      jdeErrorSet (lpBhvrCom, lpVoid, (ID) 0, _J("4363"), (LPVOID) NULL);
      return ER_ERROR;
    }

    idJDEDBReturn = JDB_InitBhvr(lpBhvrCom, &hUser, (JCHAR *) NULL,
	                              JDEDB_COMMIT_AUTO);

	if (idJDEDBReturn == JDEDB_FAILED)
	{
      jdeSetGBRError ( lpBhvrCom, lpVoid, (ID)0, _J("3143"));
      return ER_ERROR;
	}

   /************************************************************************
    * Set pointers
    ************************************************************************/

   /************************************************************************
    * Main Processing
    ************************************************************************/

	lpDS->cErro = _J('0');
    jdeStrcpy((JCHAR *)lpDS->szDescricaoErro,_J(" "));

	if (lpDS->idLista != 0) {
      lpIndiceLista = (LPLINKLIST)jdeRetrieveDataPtr(hUser, 
	                                      (unsigned long)lpDS->idLista);

	  if (lpIndiceLista != (LPLINKLIST)NULL) {
		   lpDadosLista = (void *)jdeAlloc(COMMON_POOL,(unsigned long)sizeof(Nodo),LMEM_ZEROINIT );
		   memset ((void *)(lpDadosLista), (int)(_J('\0')) , sizeof(Nodo));
		   //
		   jdeStrcpy((JCHAR *)(lpDadosLista->szChave),(const JCHAR *)(lpDS->szChave));
		   jdeStrcpy((JCHAR *)(lpDadosLista->szCampo1),(const JCHAR *)(lpDS->szCampo1));
		   jdeStrcpy((JCHAR *)(lpDadosLista->szCampo2),(const JCHAR *)(lpDS->szCampo2));
		   jdeStrcpy((JCHAR *)(lpDadosLista->szCampo3),(const JCHAR *)(lpDS->szCampo3));
		   jdeStrcpy((JCHAR *)(lpDadosLista->szCampo4),(const JCHAR *)(lpDS->szCampo4));
		   jdeStrcpy((JCHAR *)(lpDadosLista->szCampo5),(const JCHAR *)(lpDS->szCampo5));
		   //
	       dsItemLista.ptrData = lpDadosLista;
           dsItemLista.SortKey = (void *)NULL;
           //
	       AddLinkListItem ( lpIndiceLista,&dsItemLista,
                             sizeof(Nodo));
            //
	     	if (lpIndiceLista != (LPLINKLIST)NULL) {
     			//Armazena a Lista 
			    lpDS->idLista =  (ID)jdeStoreDataPtr(hUser, (void *)lpIndiceLista);
			} else {
	     	   lpDS->cErro = _J('5');
			   jdeStrcpy((JCHAR *)lpDS->szDescricaoErro,_J("Não conseguiu inserir o item na lista"));
			};

	  } else {
     	   lpDS->cErro = _J('4');
		   jdeStrcpy((JCHAR *)lpDS->szDescricaoErro,_J("Não conseguiu restaurar a lista"));
	  }
	} else {
	   lpDS->cErro = _J('3');
       jdeStrcpy((JCHAR *)lpDS->szDescricaoErro,_J("Id Nulo"));
	}

   /************************************************************************
    * Function Clean Up
    ************************************************************************/
    if(hUser != (HUSER)NULL)
    {
       JDB_FreeBhvr(hUser);
    }

    return (ER_SUCCESS);
}

		//pega o primeiro registro
		GetFirstLinkListItem(lpIndiceLista, &dsItemLista);
		while ((Retorno == TRUE)  && (Para != _J('S'))) {
			//
            lpDadosLista = lpIndiceLista->ptrCurrentItem->ptrData;
            //
			if (jdeStrcmp((JCHAR *)(lpDS->szChave), (JCHAR *)(lpDadosLista->szChave)) == 0) {
				jdeStrcpy((JCHAR *)(lpDS->szChave), (const JCHAR *)(lpDadosLista->szChave));
	    		jdeStrcpy((JCHAR *)(lpDS->szCampo1),(const JCHAR *)(lpDadosLista->szCampo1));
				jdeStrcpy((JCHAR *)(lpDS->szCampo2),(const JCHAR *)(lpDadosLista->szCampo2));
				jdeStrcpy((JCHAR *)(lpDS->szCampo3),(const JCHAR *)(lpDadosLista->szCampo3));
				jdeStrcpy((JCHAR *)(lpDS->szCampo4),(const JCHAR *)(lpDadosLista->szCampo4));
				jdeStrcpy((JCHAR *)(lpDS->szCampo5),(const JCHAR *)(lpDadosLista->szCampo5));
				lpDS->cErro = _J('S');//Achou o registro
				Para = _J('S');
			} else {
				//vai avançando registro a registro
				Retorno = GetNextLinkListItem(lpIndiceLista, &dsItemLista);
				lpDadosLista = lpIndiceLista->ptrCurrentItem->ptrData;
				//
			}
		  //
		}
		//
		lpDS->idLista = (ID)jdeStoreDataPtr(hUser, (void *)lpIndiceLista);
		//
	} else {
        lpDS->cErro = _J('7');
	    jdeStrcpy((JCHAR *)(lpDS->szDescricaoErro),(const JCHAR *)(_J("ID Nulo")));
	}
   /************************************************************************
    * Function Clean Up
    ************************************************************************/

   return (ER_SUCCESS);
}
