int I57LOGEX_CarregaArquivoMemoria(const ZCHAR *NomeArquivo, ZCHAR **Retorno) 
{ 
	int Tamanho = 0;
	FILE *Arquivo = fopen(NomeArquivo, "rb");
	if (Arquivo == NULL) 
	{ 
		*Retorno = NULL;
		return -1; 
	} 
	fseek(Arquivo, 0, SEEK_END);
	Tamanho = ftell(Arquivo);
	fseek(Arquivo, 0, SEEK_SET);
	*Retorno = (ZCHAR *)malloc(Tamanho+1);
	if (Tamanho != (int)fread(*Retorno, sizeof(ZCHAR), Tamanho, Arquivo)) 
	{ 
		free(*Retorno);
		return -2; 
	} 
	fclose(Arquivo);
	(*Retorno)[Tamanho] = 0;
	return Tamanho;
}
