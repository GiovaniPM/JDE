       BF mnE [LUSD] = "0"

  End If
  //
  // ***** Desconsiderar o E caso nenhum campo for selecionado *****
 
-
If BF mnVR [LUSD] is equal to "0"
  And BF mnVO [LUSD] is equal to "0"
  And BF mnDA [LUSD] is equal to "0"

       BF mnE [LUSD] = "0"
       BF mnOU [LUSD] = "1"

  End If
  //
  // ***** Soma os valores dos campos, com o valor LinhaAnterior *****
  BF mnVerificaLinhaAnt [LUSD] = ([BF mnVerificaLinhaAnt [LUSD]]+([BF mnVR [LUSD]]+[BF mnVO [LUSD]]))+[BF mnDA [LUSD]]
  //
 
-
If BF mnNivel [LUSD] is equal to "1"

       // VR ou VO ou DA
       //
       // ***** Calcula o resultado da linha atual do Nivel 1 *****
       VA evt_ResultAtual [LUSD] = (([BF mnVR [LUSD]]*([BF szMVR [CCD0]]+[BF mnZerado [LUSD]]))+([BF mnVO [LUSD]]*([BF szMVO [CCD0]]+[BF mnZerado [LUSD]])))+([BF mnDA [LUSD]]*([BF szMDA [CCD0]]+[BF mnZerado [LUSD]]))
       //

 
-
Else

       // VR e VO e DA
       //
       // ***** Calcula o resultado da linha atual do Nivel 2 *****
       VA evt_ResultAtual [LUSD] = (((([BF mnVR [LUSD]]-1)+([BF mnVR [LUSD]]*([BF szMVR [CCD0]]+[BF mnZerado [LUSD]])))*(([BF mnVO [LUSD]]-1)+([BF mnVO [LUSD]]*([BF szMVO [CCD0]]+[BF mnZerado [LUSD]]))))*(([BF mnDA [LUSD]]-1)+([BF mnDA [LUSD]]*([BF szMDA [CCD0]]+[BF mnZerado [LUSD]]))))*(([BF mnVR [LUSD]]+[BF mnVO [LUSD]])+[BF mnDA [LUSD]])
       //

  End If
  //
  // ***** Calcula o resultado atual e resultado anterior com E e OU *****
  BF mnResultAnterior [LUSD] = (([BF mnResultAnterior [LUSD]]*[VA evt_ResultAtual [LUSD]])*[BF mnE [LUSD]])+(([BF mnResultAnterior [LUSD]]+[VA evt_ResultAtual [LUSD]])*[BF mnOU [LUSD]])
  //
  // ***** Se Resultado Anterior for diferente de zero, será 1*****
 
-
If BF mnResultAnterior [LUSD] is not equal to "0"

       BF mnResultAnterior [LUSD] = "1"

  End If
